    city, 
    COUNT(DISTINCT user_id) AS users,
    AVG(read_time) AS avg_read
  FROM wechat_data
  WHERE last_active_date > CURDATE() - INTERVAL 7 DAY
  GROUP BY city
  ORDER BY users DESC
  LIMIT 10
